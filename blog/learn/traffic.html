    # 模拟交通灯系统

    # 要求

    1. 网页上红, 绿, 黄三种颜色交替展示
    2. 每种颜色停留的时间是固定的数值, 比如绿灯10s, 黄灯3s, 红灯5s
    3. 同时展示倒计时
    4. 顺序: 红色 -> 黄色 -> 绿色
    5. 循环往复不能停止
    6. 选做: 有可能可以设置启动, 停止, 恢复操作

    # 思路

    1. 使用promise+async await达到等待的效果
    2. 使用requestAnimationFrame 达到颜色刷新的效果。这里不使用setTimeout 和 setInterVal 的原因, 可具体查阅一下这三个的区别及后两个的缺点。

    # 实现

    <!DOCTYPE html>
<html lang="en">
<head>
    <meta charset="UTF-8">
    <meta name="viewport" content="width=device-width, initial-scale=1.0">
    <title>Document</title>
    <style>
        #some-element-you-want-to-animate {
            width: 50px;
            height: 50px;
            background-color: red;
            border-radius: 50%;
        }
    </style>
</head>
<body>
    <div id="some-element-you-want-to-animate"></div>
    <button id="start">开始</button>
    <button id="end">暂停</button>
    <button id="regain">恢复</button>
    <p id="time"></p>
</body>
<script>
    const element = document.getElementById('some-element-you-want-to-animate'); 
    const startBtn = document.getElementById('start');
    const timeDesc = document.getElementById('time');
    const endBtn = document.getElementById('end');
    const regainBtn = document.getElementById('regain');

    let lastTime = 0; // 上一次的时间
    let timer = null;
    let currentColor = '';
    let remainTime = 0;
    let colorList = ['red','yellow','green'];
    let timeMap = {
        'red': 5,
        'yellow': 3,
        'green': 10
    }

    startBtn.addEventListener('click',start);
    endBtn.addEventListener('click',end);
    regainBtn.addEventListener('click',regain);

    async function start(index, time) {
        let i = isNaN(parseInt(index)) ? 0 : index; // 兼容点击开始按钮时 index非数字的情况
        for(;i<colorList.length;i++) {
            let currentTime = timeMap[colorList[i]]; // 获取map中对应的灯时
            if(i === index) { //  i=== index 可能是恢复的情况, 所以需要将currentTime改变为恢复时还剩余的时间
                currentTime =  (time ? time : timeMap[colorList[i]]);
            }
            await turnOnLight(colorList[i],currentTime)
        }
        await start(0) // 每次循环时从第一个灯开始
    };

    function end() {
        cancelAnimationFrame(timer);
    }

    function regain() {
        const index = colorList.findIndex(item => item === currentColor);
        start(index, remainTime)
    }

    function turnOnLight(color,time) {
        cancelAnimationFrame(timer);
        return new Promise(resolve =>  {
            timer = requestAnimationFrame(function callback() {
                if(Date.now() - lastTime >= 1000){ // 如果上次时间和当前时间差大于1s 就需要更新页面的展示
                    if(time) { // 如果当前灯时还未结束
                        lastTime = Date.now();
                        timer = requestAnimationFrame(callback);
                        timeDesc.innerHTML = `倒计时: ${time}`;
                        element.style.backgroundColor = color;
                        currentColor = color;
                        remainTime = time;
                        time -= 1;
                    }else { // 如果当前灯时结束 那么就清空动画 并将promise状态置为成功
                        cancelAnimationFrame(timer);
                        resolve();
                    }
                }else {
                    timer = requestAnimationFrame(callback)
                }
            })
        })
    }
</script>
</html>
